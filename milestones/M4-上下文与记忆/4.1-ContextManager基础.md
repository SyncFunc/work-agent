# agent/context/manager.py


@dataclass
class ContextUsage:
    """上下文占用分类明细。"""

    system_fixed: int = 0  # System Prompt 静态段（可缓存）
    system_dynamic: int = 0  # System Prompt 动态段（日期/Git 状态等）
    tools: int = 0  # Tools 列表
    messages: int = 0  # 对话历史（user/assistant/tool_result）
    total: int = 0  # 总和
    available: int = 0  # 有效窗口 − total（剩余可用）
    used_pct: float = 0.0  # total / 有效窗口


@dataclass
class CompactRecord:
    """一次压缩的记录。"""

    ts: float  # 时间戳
    method: str  # "microcompact" / "auto_compact" / "session_memory"
    before_tokens: int
    after_tokens: int


class ContextManager:
    def __init__(
        self,
        context_window: int = 200_000,
        max_output_tokens: int = 20_000,
        compact_buffer: int = 13_000,
        *,
        system_fixed_tokens: int = 3_000,
        system_dynamic_tokens: int = 0,
        tools_tokens: int = 15_000,
    ):
        self.conv: list[Message] = []
        self.context_window = context_window
        self.max_output_tokens = max_output_tokens
        self.compact_buffer = compact_buffer
        self.effective_window = context_window - min(max_output_tokens, 20_000)
        self.compact_threshold = self.effective_window - compact_buffer
        self.compact_boundary: int = 0  # 索引边界：之前的内容已被压缩
        self.history: list[CompactRecord] = []
        self._system_fixed = system_fixed_tokens
        self._system_dynamic = system_dynamic_tokens
        self._tools = tools_tokens

    def estimate_usage(self) -> ContextUsage:
        """估算当前上下文占用分类明细。"""
        msg_tokens = self._estimate_conv_tokens()
        fixed = self._system_fixed
        dynamic = self._system_dynamic
        tools = self._tools
        total = fixed + dynamic + tools + msg_tokens
        available = max(0, self.effective_window - total)
        used_pct = total / self.effective_window if self.effective_window > 0 else 0
        return ContextUsage(fixed, dynamic, tools, msg_tokens, total, available, used_pct)

    def should_compact(self) -> bool:
        """是否需要触发压缩？"""
        usage = self.estimate_usage()
        return usage.used_pct >= 0.93  # 约 ~187K/180K（200K 窗口）

    def mark_boundary(self) -> None:
        """记录当前 conv 索引为压缩边界。"""
        self.compact_boundary = len(self.conv)

    def _estimate_conv_tokens(self) -> int:
        """估算 conv 中 boundary 之后的 token 数。"""
        text = ""
        for msg in self.conv[self.compact_boundary :]:
            text += (msg.content or "") + (msg.name or "")
            if msg.tool_calls:
                for tc in msg.tool_calls:
                    text += tc.name + json.dumps(tc.arguments, sort_keys=True)
        return _estimate_tokens(text)

    def set_conv(self, conv: list[Message]) -> None:
        """设置当前上下文投影（每次 loop.run 前调用）。"""
        self.conv = conv

    def get_active_messages(self) -> list[Message]:
        """获取 boundary 之后的有效消息（压缩前原文 + 压缩后消息）。"""
        return list(self.conv)

## 知识沉淀

> M4.1 已完成。关键接口、公式与踩坑记录如下（供 M4.2–M4.7 复用）。

### 接口签名与默认值（落到 `agent/context/manager.py`）

- `ContextManager(context_window=200_000, max_output_tokens=20_000, compact_buffer=13_000, *, system_fixed_tokens=3_000, system_dynamic_tokens=0, tools_tokens=15_000)`。
- 关键属性：`conv: list[Message]`、`effective_window`、`compact_threshold`、`compact_boundary: int`、`history: list[CompactRecord]`。
- 方法：`estimate_usage() -> ContextUsage`、`should_compact() -> bool`、`mark_boundary()`、`record_compact(method, before, after) -> CompactRecord`、`set_conv(conv)`、`get_active_messages() -> list[Message]`、`_estimate_conv_tokens()`（私有）。
- `ContextUsage` 字段：`system_fixed/system_dynamic/tools/messages/total/available/used_pct`。
- `CompactRecord` 字段：`ts/method/before_tokens/after_tokens`。

### 公式（与 Claude Code 对齐）

- 有效窗口：`effective_window = context_window − min(max_output_tokens, 20000)`（保留最大输出预算，避免输出被截断）。默认 `200_000 − 20_000 = 180_000`。
- 压缩阈值：`compact_threshold = effective_window − compact_buffer`（默认 `180_000 − 13_000 = 167_000`）。
- 触发条件：`should_compact()` 在 `used_pct >= 0.93` 时返回 True（≈ `effective_window − compact_buffer`，与 `compact_threshold` 一致）。

### 关键决策 / 踩坑

- `_estimate_tokens` 从 `agent/runtime/terminal_transport.py` **抽取到 `agent/context/tokens.py` 共享**，并由 `terminal_transport` 反向 import 复用（单一事实来源，避免重复实现，也避免 `context → runtime` 的循环依赖）。`tokens.py` 不依赖任何其它 agent 模块，零循环导入风险。
- **`Message` 无 `name` 字段**（文档 4.1 草稿里的 `msg.name` 是笔误）：`_estimate_conv_tokens` 改用 `msg.tool_call_id` 计入配对标识，`msg.tool_calls` 用 `tc.name + json.dumps(tc.arguments, sort_keys=True, ensure_ascii=False)` 估算。
- 计量只统计 `conv[self.compact_boundary:]`（boundary 之后），压缩区不重复计量。
- `record_compact()` 是 4.1 草稿未列、但验收要求「历史记录」所需的方法，已补充（用于 M4.2+ 压缩时落记录）。
- `Compactor` 协议定义在 `agent/context/compactors/__init__.py`（`runtime_checkable` Protocol，`async compact(conv, boundary) -> list[Message]`）；具体实现 Microcompact / Auto Compact / Session Memory 在 M4.2–M4.4 落地，本步只定义协议。
- 本步**只做计量与边界管理**，不执行压缩；`ContextManager` 不触碰 `EventStream`（审计真相不可变）。

### 配置（`agent/config/settings.py` 新增 `ContextConfig`）

- `Settings.context: ContextConfig`，默认 `context_window=200_000 / max_output_tokens=20_000 / compact_buffer=13_000 / microcompact_keep_recent=5 / microcompact_enabled=True / auto_compact_enabled=True / session_memory_enabled=True / session_memory_dir=".agent/sessions" / agents_md_path="AGENTS.md" / agents_md_enabled=True`。
- `ContextManager` 当前由显式参数构造（未强绑定 `Settings`）；M4.5 集成时可由 `settings.context` 派生阈值。
