**关键测试策略（对齐项目「LLM 一律可 Mock」）**：

```python
# 子 agent 上下文隔离（核心不变量）
def test_subagent_context_isolated():
    base = FakeModel([Decision(text="child final")])
    spawner = SubagentSpawner(settings, max_depth=5)
    spec = BUILTIN_GENERAL
    result = asyncio.run(spawner.spawn(spec, "do X", base_model=base))
    # 子 messages 从 [] 起，父看不到子内部
    assert result.text == "child final"
    # 父 conv 不污染：spawn 不修改调用方 messages
```

# Skill 正文按需加载（省 token 不变量）

```python
# Trace 父子（不 reset 到根）
def test_subagent_span_parented():
    tracer = Tracer()
    base = FakeModel([Decision(text="x")])
    spawner = SubagentSpawner(settings, tracer=tracer)
    parent = tracer.span("agent.run")
    with parent:
        result = asyncio.run(spawner.spawn(BUILTIN_GENERAL, "t",
                              depth=0, parent_span=parent, base_model=base))
    # 子 model.act span.parent_id == parent.id
```
